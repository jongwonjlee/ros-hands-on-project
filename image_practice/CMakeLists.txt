cmake_minimum_required(VERSION 2.8.3)
project(image_practice)

## Compile as C++11, supported in ROS Kinetic and newer
add_compile_options(-std=c++11)

## Find catkin macros and libraries
## if COMPONENTS list like find_package(catkin REQUIRED COMPONENTS xyz)
## is used, also find other catkin packages
find_package(catkin REQUIRED COMPONENTS
  cv_bridge
  image_transport
  roscpp
  sensor_msgs
)

# Asks CMake to find and load settings from the denoted external package.
# CMake searches the directories listed in CMAKE_MODULE_PATH for a file called Find<name>.cmake.
find_package( cv_bridge REQUIRED )

# Specify catkin-specific information to the build system which in turn is used to generate pkg-config and CMake files.
# INCLUDE_DIRS - The exported include paths (i.e. cflags) for the package
# LIBRARIES - The exported libraries from the project
# CATKIN_DEPENDS - Other catkin projects that this project depends on
# DEPENDS - Non-catkin CMake projects that this project depends on. For a better understanding, see this explanation.
catkin_package(
  INCLUDE_DIRS include
#  CATKIN_DEPENDS roscpp
#  sensor_msgs
#  std_msgs
#  image_transport
#  DEPENDS system_lib
)

# The argument to include_directories should be the *_INCLUDE_DIRS variables generated by your find_package calls
# and any additional directories that need to be included.
include_directories(
  ${cv_bridge_INCLUDE_DIRS}
  ${catkin_INCLUDE_DIRS}
  #include
)

# Specify an executable target that must be built
add_executable(image_republisher src/image_republisher.cpp)
# Specify which libraries an executable target links against
target_link_libraries(image_republisher   ${catkin_LIBRARIES} ${cv_bridge_LIBRARIES})
